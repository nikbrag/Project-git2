1) # Узнаем установленную версию Git
git --version
# git version 2.36.1.windows.1
2) # Иницилизируем Git внутри проекта Project Git
git init
# Initialized empty Git repository in D:/ItProger/Project Git/.git/
3) # выполним настройку account и присваиваем имя пользователя
git config --global user.name 'Nikita Bragin'
# присваиваем имя почты
git config --global user.email 'nikbrag@gmail.com'
4) # добавляем все файлы в локальный репозиторий - сначала в статусе ожидания
git add *.html
# добавляем все файлы у которых расширение html

git add css/
# добавляем папку css

git add css/*.scss
# добавляем все файлы с расширением .scss из папки css
# /some/**/*.txt  - внутри папки some беру все папки и игнирирую все файлы с расширением txt

5) # сначала загрузим index.html
git add index.html

# проверяем статус всех фалов
git status

# отменяем загрузку файла index.html
git rm --cached index.html

git status
# показывает что нет файлов в статусе ожидания загрузки

6) # добавляем полностью все файлы на загрузку

git add .
git status

7) # выполняем загрузку
# так как сделали больше изменений то снова пишем
git add .
git commit

# Попадаем в редактор Win

# Нажимаем на <I>
# Появляется Insert
# пишем в win
Initial commit

# или удаляем # около Initial commit
# нажимаем Esc и прописываем в самом конце
:wq

# происходит commit и данные заносятся в репозиторий
git status

git add .

# чтобы обойти редактор win
git commit -m 'Change test_11.txt'

git status

8) # для игнорирования создаем новый файл .gitignore
# внути файла указываем какие файлы и папки мы игнорируем
# указываем папки
/bin
/admin
/config
*.sass
